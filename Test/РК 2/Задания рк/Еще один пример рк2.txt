Билет N 15. С клавиатуры вводится последовательность чисел через пробел. 
Необходимо найти число, которое встретилось максимальное число раз, и вывести его на печать. 
Если таких чисел несколько, взять наименьшее.

array = [1, 2, 4, 3, 1, 1, 1, 5, 0, 0, 0, 0, -1]

value, count = array
    .group_by {|element| element }
    .map {|key, value| [key, value.length] }
    .sort
    .sort_by {|key, value| -value }[0]

puts value, count
